# 数据库管理

# Linux服务器环境

切换 Oracle 用户

su - oracle

进入 sqlplus

sqlplus /nolog 
conn /as sysdba

# 启动数据库

进入 sqlplus 后

startup

# 关闭数据库

进入 sqlplus 后

shutdown immediate

# 数据库监听

启动监听

lsnrctl start

查看监听状态

lsnrctl status
