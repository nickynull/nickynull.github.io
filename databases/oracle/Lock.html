# 锁表处理

# 查看锁

查询锁表信息1
------------------------------------------
--查询锁表信息
------------------------------------------
SELECT 
  LOC.SESSION_ID || ',' || SESS.SERIAL# KEY, 
  LOC.SESSION_ID      SID, 
  SESS.SERIAL#, 
  LOC.LOCKED_MODE     锁模式, 
  LOC.ORACLE_USERNAME 登录用户, 
  LOC.OS_USER_NAME    登录机器用户名, 
  SESS.PROGRAM        软件,
  SESS.MACHINE        机器名, 
  SESS.TERMINAL       终端用户名, 
  OBJ.OBJECT_NAME     被锁对象名, 
  SESS.LOGON_TIME     登录数据库时间,
  LOC.LOCKED_MODE     锁定模式,
  SESS.LOCKWAIT       等待,
  'ALTER SYSTEM KILL SESSION ''' || loc.SESSION_ID || ',' || sess.SERIAL# || ''';' AS 强制结束会话语句
FROM V$LOCKED_OBJECT LOC
  LEFT JOIN ALL_OBJECTS OBJ ON LOC.OBJECT_ID = OBJ.OBJECT_ID 
  LEFT JOIN V$SESSION SESS ON LOC.OBJECT_ID = OBJ.OBJECT_ID AND LOC.SESSION_ID = SESS.SID 
WHERE 1=1;
查询锁表信息2
SELECT lo.inst_id,
       lo.oracle_username AS ora_user,
       lo.os_user_name AS os_user,
       lo.process,
       lo.locked_mode,
       do.owner,
       do.object_name,
       s.sid,
       s.serial#,
       'alter system kill session ''' || s.sid || ',' || s.serial# || ',@' ||
       s.inst_id || ''';' AS kill_ddl
  FROM gv$locked_object lo, dba_objects do, gv$session s
 WHERE lo.object_id = do.object_id
   AND lo.session_id = s.sid
   AND lo.inst_id = s.inst_id
   AND do.owner = 'OWNER'
   AND do.object_name = 'TEST230317';
查询锁表记录
------------------------------------------
--表记录 锁定查询
------------------------------------------
SELECT 
  LOC.*,
  DOBJ.OWNER, DOBJ.OBJECT_NAME,
  'ALTER SYSTEM KILL SESSION ''' || LOC.SESSION_ID || ',' || SESS.SERIAL# || ''';'
FROM GV$LOCKED_OBJECT LOC
  LEFT JOIN DBA_OBJECTS DOBJ ON LOC.OBJECT_ID = DOBJ.OBJECT_ID
  LEFT JOIN GV$SESSION SESS ON LOC.SESSION_ID = SESS.SID AND LOC.SESSION_ID = SESS.SID
WHERE 1=1
  AND DOBJ.OBJECT_NAME = &表名;

# 查某SESSION正在执行的SQL语句

查某SESSION 正在执行的SQL语句，从而可以快速定位到哪些操作或者代码导致事务一直进行没有结束等.

------------------------------------------
--查某SESSION 正在执行的SQL语句，从而可以快速定位到哪些操作或者代码导致事务一直进行没有结束等.
------------------------------------------
SELECT /*+ ORDERED */
  SQL_TEXT
FROM V$SQLTEXT TX
WHERE (TX.HASH_VALUE, TX.ADDRESS) IN (
  SELECT 
    DECODE(SQL_HASH_VALUE, 0, PREV_HASH_VALUE, SQL_HASH_VALUE),
    DECODE(SQL_HASH_VALUE, 0, PREV_SQL_ADDR, SQL_ADDRESS)
  FROM V$SESSION SESS
  WHERE SESS.SID = &SID
) /* 此处为SID*/
ORDER BY PIECE ASC;

# 查看操作系统PID

------------------------------------------
--查看操作系统PID
------------------------------------------
SELECT 
  SPID,OSUSER
  ,SESS.PROGRAM 
FROM V$SESSION SESS
  LEFT JOIN V$PROCESS PRO ON SESS.PADDR=PRO.ADDR 
WHERE 1=1
  AND SESS.SID = &SID;
