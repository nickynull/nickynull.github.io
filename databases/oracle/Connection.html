# 数据库连接

# 查看数据库连接数

SELECT t.resource_name       res_name,
       t.current_utilization "当前连接数",
       t.max_utilization     "最大连接数",
       t.initial_allocation  "设置的连接数",
       t.limit_value         limit_value
  FROM gv$resource_limit t
 WHERE t.resource_name IN ('processes', 'sessions');

# 修改数据库最大连接数

提醒

修改数据库最大连接数需要重启数据库

su - oracle #使用root登录后，切换到oracle用户，或者直接登录到oracle用户
sqlplus / as sysdba

alter system set processes=5000 scope=spfile sid='*';
shutdown immediate
startup
